052
053    @Nonnegative
054    public int getCompressionOverhead(@Nonnegative int inputBufferSize);

056    public int getCompressionLevel();
057
058    @Nonnegative
059    public int compress(@Nonnull byte[] in, @Nonnegative int in_base, @Nonnegative int in_len,
060            @Nonnull byte[] out, @Nonnegative int out_base, @Nonnull lzo_uintp out_len);
